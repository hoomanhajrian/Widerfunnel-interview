var jqry = document.createElement('script');
jqry.src = "https://code.jquery.com/jquery-3.3.1.min.js";
document.getElementsByTagName('head')[0].appendChild(jqry);
jQuery.noConflict();

//Defining objects for later use

const button = $("<button id='modalButton'>Modal</button>");

const modal = $(`<div id="modal" style='display:none;width:100%;height:100%;background-color:rgb(0,0,0);background-color:rgb(0,0,0,0.4);position:fixed;top:0;left:0;z-index:99;overflow:'auto';'><div style='background-color:#fefefe;margin:15% auto;padding: 20px;border:1px solid #888;width:80%;' class='modal-overlay'><div class='modal-content'><button id="modalCloseButton" style='background:transparent;border:none;float:right;'>Ã—</button><h1>Modal</h1><p>This is an example text for the modal.</p></div></div></div>`);

//Assigning objects to variables by id or className

sidebarContainer = $("#sidebar");
sidebar = $("#sidebar").children();
sidebar.append(button);
modalButton = $("#modalButton");

//we are getting the container distance to top and whole height to be able to control the scroll
//with the figures availabel

let a = sidebarContainer.offset().top - 100;
let b = sidebarContainer.height();

$(window).scroll(() => {
    let scrollYpos = $(document).scrollTop();
    if(a<scrollYpos && scrollYpos < b + 600){
       sidebar.css({
            'top': '100px',
            'position': 'sticky'
        });
    }
    else{
         sidebar.css({
            'top': 'auto',
            'position': 'relative'
        });
    }
});
$(document.documentElement).append(modal); 
modalAttr = $("#modal");
modalCloseButton = $("#modalCloseButton");

//event handlers for modal

modalButton.click(()=>{
    modalAttr.css("display","inline");
});

modalCloseButton.click(()=>{
   modalAttr.css("display","none"); 
});
